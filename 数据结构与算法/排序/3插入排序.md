### 代码
```js
function insertSort(arr) {
  if (arr.length < 2) return;

  // i 是 为排序的元素index
  for (let i = 1; i < arr.length; i++) {
    // j是 已经排序的列表index
    for (let j = i - 1; j >= 0 && arr[j] > arr[j + 1]; j--) {
      [arr[j + 1], arr[j]] = [arr[j], arr[j + 1]];
    }
  }
}
```
