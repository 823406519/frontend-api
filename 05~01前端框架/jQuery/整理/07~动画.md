
## 动画
```js
/*
speed：Default value is 400 milliseconds
easing ：Default value is "swing"
*/
$(selector).show(speed,easing,callback) // 改变display，显示元素
$(selector).hide(speed,easing,callback) // 相当于display:none in CSS

$(selector).fadeIn(speed,easing,callback) // 改变opacity，显示元素
$(selector).fadeOut(speed,easing,callback) // 改变opacity，隐藏元素
$(selector).fadeToggle(speed,easing,callback)

$(selector).slideDown(speed,easing,callback) // slideown 显示元素
$(selector).slideUp(speed,easing,callback) // 隐藏元素


(selector).animate({styles},speed,easing,callback)
(selector).animate({styles},{options})

// 停止当前动画
// isStopAll为true则停止所有发生在selector的动画，默认为false
// isGotoEnd指示是否立即完成所有动画，默认为false
$(selector).stop(isStopAll, isGoToEnd)
```
