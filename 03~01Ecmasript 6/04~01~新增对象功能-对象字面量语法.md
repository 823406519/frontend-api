## 对象属性的简写
```js
// 省略冒号和值
function createPerson(name, age){
    return {
        name,
        age
    }
}
```

## 对象方法简写
```js
// 省略冒号和function
let person = {
    name: 'liuyaohui',
    sayName(){
        console.log(this.name);
    }
}
```

## 对象计算属性
```js
// 使用对象计算属性访问
let person = {};
firstName = 'first name';
person[firstName] = 'liu';
person['first name'] // 'liu'

// 字面量使用对象属性访问
let firstName = 'first name';
let person = {
    [firstName]: 'liu'
}
person['first name'] // 'liu'

// 对象计算属性也可以使用表达式
let person = {};
let first ='first';
person[first + ' name'] = 'liu';
person['first name']; // liu

```
