
## 剩余项解构
剩余项是数组解构的最后部分，之后不能有逗号，否则出现语法错误
```js
let colors = ['red', 'green', 'blue'];
[firstColor, ...restColors] = colors;
console.log(firstColor, restColors); // red [ 'green', 'blue' ]

/*
=== ES5中通常采用concat复制一个新数组
*/
let colors = ['red', 'green', 'blue'];

newColors = colors.concat();
console.log(newColors); // [ 'red', 'green', 'blue' ]
// === ES6中使用剩余项可以复制一个新数组
// 剩余项是数组解构的最后部分，之后不能有逗号，否则出现语法错误
let colors = ['red', 'green', 'blue'];
[...newColors] = colors;
console.log(newColors); // [ 'red', 'green', 'blue' ]

```
