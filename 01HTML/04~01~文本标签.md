
```html
<!-- 标题: h1-h6 -->
<h1>title</h1>

<!-- 段落 -->
<p>paragraph</p>

<!-- 粗体和斜体 -->
<b>bold</b>
<i>italic</i>

<!-- 上下标 -->
<sup>上标</sup>
<sub>下标</sub>

<!-- 换行和水平线 -->
<br>
<hr>

<!-- 语义化标签 -->
<strong>强调</strong>
<em>强调</em>

<!-- 引用 -->
<blockquote>块引用</blockquote>
<q>短引用</q>

<!-- 引文和定义 -->
<cite>常用语作品引用的标记</cite>
<dfn>术语的解释</dfn>

<!-- 缩写 -->
<abbr title="缩写的详述">缩写</abbr>

<!-- <address> 元素 表示其中的 HTML 提供了某个人或某个组织（等等）的联系信息 -->
<address>
  <a href="mailto:jim@rock.com">jim@rock.com</a><br>
  <a href="tel:+13115552368">(311) 555-2368</a>
</address>

<!-- 内容修改 -->
<ins>增加的内容</ins>
<del>已经删除的文本</del>
<s>已经删除的文本</s>

<!-- 代码 -->
<code>文本中的短代码</code>
<pre>一段代码</pre>

<!-- small -->
<small>更小的字体</small>

<!-- <time> 标签表示24小时制时间或者公历日期 -->
<!-- 以下显示结果为The game starts at 19:00. -->
The game starts at <time datetime="2017-04-29T19:00">19:00</time>.

<!-- 为了突出指定内容， 标记高亮显示文字 -->
<mark></mark>
<!-- 键盘输入元素(<kbd>) 用于表示用户输入，它将产生一个行内元素，以浏览器的默认monospace字体显示 -->
<kbd>Ctrl</kbd> + <kbd>S</kbd>

<!-- <ruby> 元素 被用来展示东亚文字注音或字符注释 -->
<!-- <rp> 元素用于为那些不能使用 <ruby> 元素展示 ruby 注解的浏览器，提供随后的圆括号 -->
<ruby>
  汉 <rp>(</rp><rt>han</rt><rp>)</rp>
  字 <rp>(</rp><rt>zi</rt><rp>)</rp>
</ruby>

<!-- <bdo> 元素 (HTML双向覆盖元素)用于覆盖当前文本的朝向，它使得字符按给定的方向排列 -->
<!-- dir : ltr | rtl -->
<bdo dir="rtl">This text will go right to left.</bdo>
```
